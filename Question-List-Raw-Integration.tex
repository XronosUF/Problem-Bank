    \[
    \int \sage{Integrand(x)} dx.

        \[
        \begin{array}{lr}
        \sage{radicand(x)}= (\answer{\sage{rad1}})^2 + \answer{\sage{rad2}} 
        \end{array}

        \choice{$\sage{fake2}$}

        \[
        \begin{array}{lr}
        a = \answer{\sage{b1}} 
        \end{array}

        \end{array}

        \]
\begin{problem}

        \[
        \int  \sage{Integrand(x)} dx  = \answer[validator=sameDerivative]{\sage{Ans} + C}
        \]
\end{problem}
\end{problem}

\end{problem}
